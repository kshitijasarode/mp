      T1M1Delay();

void T1M1Delay (void)
{
TMOD=0x10; //Timer 1, delay 1 (16 bit)
TL1=0xFE; //Load TL1
TH1=0xA5; //Load TH1
TR1=1; // Turn on T1
while(TF1==0); // Wait for TF1 to roll over
TR1=0; // Turn off T1
TF1=0; // Clear TF1
}
